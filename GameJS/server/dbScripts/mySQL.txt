CREATE DATABASE IF NOT EXISTS `testdatabase` /*!40100 DEFAULT CHARACTER SET utf8 */;


CREATE TABLE IF NOT EXISTS `cubes` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `x` int(11) NOT NULL DEFAULT '0',
  `y` int(11) NOT NULL DEFAULT '0',
  `z` int(11) NOT NULL DEFAULT '0',
  `csId` int(11) NOT NULL DEFAULT '0',
  `csCol` int(11) NOT NULL DEFAULT '0',
  `csRow` int(11) NOT NULL DEFAULT '0',
  `tsId` int(11) NOT NULL DEFAULT '0',
  `tsCol` int(11) NOT NULL DEFAULT '0',
  `tsRow` int(11) NOT NULL DEFAULT '0',
  `created` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `modified` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`,`x`,`y`,`z`),
  UNIQUE KEY `id_UNIQUE` (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=51614 DEFAULT CHARSET=utf8;
